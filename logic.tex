\begin{flalign*}
    &\forall (y \ x) \ (z \ x) \mid \text{в $ y \ x $ стоит первым символом} \\
    &\exists x \\
    &\exists x \ (y \ x) \\
    &\exists (y \ x) \ (z \ x) \mid \text{в $ y \ x $ стоит первым символом}
\end{flalign*}
\begin{flalign*}
