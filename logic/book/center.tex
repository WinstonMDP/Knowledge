\section{Определения}
\begin{flalign*}
    &\tx{prop\_formulas} \ x
    = \\
    &\tx{ind}
    \left(
    \tx{consts} \ x
    \cup
    \set {
        \exel y
        \begin{cases}
            y \in q^q \\
            \forall z \in q \ y_{\word{z}} \equiv \word{\tx{neg}} \concat z
        \end{cases},
        \exel y
        \begin{cases}
            y \in q^q \\
            \forall z \andc w \in q \
            y_{\word{z \ w}} \equiv \word{\tx{and}} \concat z \concat w
        \end{cases}
    }
    \right) \\
    &\tx{where} \
    q = \tx{words}(x \cup \tx{prop\_connectives}) \\
    &\bb{B} = 2 \\
    &x - \tx{оценка над $ y $} = x \in \bb{B}^y \\
    &\tx{interpretation} \ x \ y \
    \fnd {
    \begin{cases}
        x \in \tx{prop\_formulas} \ z \\
        y - \tx{оценка над $ z $}
    \end{cases}
    }
    =
    \exel
\end{flalign*}
