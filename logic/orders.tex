\chapter{Порядки}
\section{Определения}
\begin{flalign*}
    &x - \tx{предпорядок}
    =
    \begin{cases}
        x - \tx{рефлексивное} \\
        x - \tx{транзитивное}
    \end{cases} \\
    &x - \tx{строгий порядок}
    =
    \begin{cases}
        x - \tx{иррефлексивное} \\
        x - \tx{транзитивное}
    \end{cases} \\
    &x - \tx{нестрогий (частичный) порядок}
    =
    \begin{cases}
        x - \tx{предпорядок} \\
        x - \tx{антисимметричное}
    \end{cases}
\end{flalign*}
\begin{flalign*}
    (x, y) - \tx{частично упорядоченное множество (ч.у.м.)}
    =
    \begin{cases}
        y - \tx{нестрогий порядок} \\
        x \equiv \cup\cup y
    \end{cases}
\end{flalign*}
\begin{flalign*}
    &x - \tx{$ y $-нижняя грань $ z $}
    =
    \begin{cases}
        y - \tx{нестрогий порядок} \\
        \forall w \in z \ (x, w) \in y
    \end{cases} \\
    &x - \tx{$ y $-верхняя грань $ z $}
    =
    \begin{cases}
        y - \tx{нестрогий порядок} \\
        \forall w \in z \ (w, x) \in y
    \end{cases}
\end{flalign*}
\begin{flalign*}
    &x - \tx{$ y $-наименьший}
    =
    x - \tx{$ y $-нижняя грань $ \tx{rng} \ y $} \\
    &x - \tx{$ y $-наибольший}
    =
    x - \tx{$ y $-верхняя грань $ \tx{rng} \ y $}
\end{flalign*}
\begin{flalign*}
    &x - \tx{$ y $-инфимум ($ y $-точная нижняя грань) $ z $}
    = \\
    &x
    -
    \tx {$
    \left(y \uparrow \set{\alpha \in z \mid \alpha - \tx{нижняя грань $ z $}}\right)
    $-наибольший
    } \\
    &x - \tx{$ y $-супремум ($ y $-точная верхняя грань) $ z $}
    = \\
    &x
    -
    \tx {$
    \left(y \uparrow \set{\alpha \in z \mid \alpha - \tx{верхняя грань $ z $}}\right)
    $-наименьший
    }
\end{flalign*}
\begin{flalign*}
    &x - \tx{$ y $-цепь}
    =
    \begin{cases}
        y - \tx{нестрогий порядок} \\
        \forall z \andc w \in x
        \left[
        \begin{aligned}
            (z, w) \in y \\
            (w, z) \in y
        \end{aligned}
        \right.
    \end{cases} \\
    &x - \tx{$ y $-антицепь}
    =
    \begin{cases}
        y - \tx{нестрогий порядок} \\
        \forall z \andc w \in x \ (z, w) \in y \to z \equiv w
    \end{cases}
\end{flalign*}
\begin{flalign*}
    x - \tx{решётка}
    =
    \begin{cases}
        x - \tx{нестрогий порядок} \\
        \forall y \andc z \in \cup\cup x
        \begin{cases}
            \exists w - \tx{$ x $-инфимум $ \set{y, z} $} \\
            \exists w - \tx{$ x $-супремум $ \set{y, z} $}
        \end{cases}
    \end{cases}
\end{flalign*}
\begin{flalign*}
    x - \tx{полная решётка}
    =
    \begin{cases}
        x - \tx{нестрогий порядок} \\
        \forall y \subseteq \cup\cup x
        \begin{cases}
            \exists z - \tx{$ x $-инфимум $ y $} \\
            \exists z - \tx{$ x $-супремум $ y $}
        \end{cases}
    \end{cases}
\end{flalign*}
\begin{flalign*}
    x - \tx{линейное}
    =
    \begin{cases}
        x - \tx{нестрогий порядок} \\
        \cup\cup x - \tx{$ x $-цепь}
    \end{cases}
\end{flalign*}
\begin{flalign*}
    &x - \tx{фундированное}
    = \\
    &\begin{cases}
        x - \tx{нестрогий порядок} \\
        \forall y \subseteq \cup\cup x \ y \not\equiv \varnothing
        \to
        \exists z \in y \
        z - \tx{$ \left(x \upharpoonright y \setminus \tx{idr} \ y\right) $-минимальный}
    \end{cases}
\end{flalign*}
\begin{flalign*}
    x - \tx{полное}
    =
    \begin{cases}
        x - \tx{линейное} \\
        x - \tx{фундированное}
    \end{cases}
\end{flalign*}
\begin{flalign*}
    x - \tx{$ y $-прогрессивное}
    =
    \begin{cases}
        y - \tx{строгий порядок} \\
        \forall z \in \tx{rng} \ y \ \left(w \to (w, z) \in y \to x \ w\right) \to x \ z
    \end{cases}
\end{flalign*}
\begin{flalign*}
    &\tx{suc} \ x \ y \
    \fnd {
    \begin{cases}
        x - \tx{полное} \\
        \exists z \ (y, z) \in x
    \end{cases}
    }
    = \\
    &\exel z
    -
    \tx{$
    \left(
    x \uparrow \set{\alpha \in \cup\cup x \mid (y, \alpha) \in x} \setminus (y, y)
    \right)
    $-наименьший} \\
    &\tx{Последователь $ y $ в $ x $}
\end{flalign*}
\begin{flalign*}
    &\begin{aligned}
        (x, y){ }-{ }&\tx{соответствие Галуа} \\
                     &\tx{между $ z $ и $ w $}
    \end{aligned}
    =
    \begin{cases}
        z \andc w - \tx{нестрогий порядок} \\
        x \in \left(\cup\cup w\right)^{\cup\cup z} \\
        y \in \left(\cup\cup z\right)^{\cup\cup w} \\
        \forall i \in \cup\cup z, j \in \cup\cup w \ (i, y_j) \in z \tot (x_i, j) \in w
    \end{cases}
\end{flalign*}
\begin{flalign*}
    x - \tx{эквивалентность}
    =
    \begin{cases}
        x - \tx{предпорядок} \\
        x - \tx{симметричное}
    \end{cases}
\end{flalign*}
\begin{flalign*}
    &\tx{ker} \ x \ \fnd{\exists y \ x - \tx{функция из $ y $}}
    =
    \set {
    (\alpha, \beta) \in \tx{zip} \left(\tx{dom} \ x\right)^2
    \mid
    x_{\alpha} \equiv x_{\beta}
    } \\
    &\tx{Ядерная эквивалентность $ x $}
\end{flalign*}
\begin{flalign*}
    &\tx{class} \ x \ y \ \fnd{y - \tx{эквивалентность}}
    =
    \set{\alpha \in \tx{dom} \ y \mid \exists z \ (x, z) \in y} \\
    &\tx{Класс эквивалентности $ x $ по $ y $}
\end{flalign*}
\begin{flalign*}
    &x / y = \set{\tx{class} \ \alpha \ y \in \mathcal{P} \ x \mid \alpha \in x} \\
    &\tx{Фактор-множество $ x $ по $ y $}
\end{flalign*}
\begin{flalign*}
    x - \tx{разбиение $ y $}
    =
    \begin{cases}
        x \subseteq \mathcal{P} \ y \\
        \cup x \equiv y \\
        \varnothing \not\in x \\
        \forall z \andc w \in x \ z \cap w \not\equiv \varnothing \to z \equiv w
    \end{cases}
\end{flalign*}

\section{Теоремы}
\begin{flalign*}
    &\set {
    \alpha \in \mathcal{P} \ \tx{zip} \ x^2 \mid \alpha - \tx{строгий порядок}
    }
    \sim
    \set {
    \alpha \in \mathcal{P} \ \tx{zip} \ x^2
    \mid
    \alpha - \tx{нестрогий порядок}
    } \\
    &x - \tx{нестрогий порядок}
    \to \\
    &\left(
    \left(
    \forall y \subseteq \cup\cup x \
    \exists z - \tx{$ x $-инфимум $ y $}
    \right)
    \tot
    \left(
    \forall y \subseteq \cup\cup x \
    \exists z - \tx{$ x $-супремум $ y $}
    \right)
    \right) \\
    &x - \tx{нестрогий порядок}
    \to \\
    &\exists y \subseteq \mathcal{P}\cup\cup x \
    (\cup\cup x, x)
    \cong
    \left(
    y,
    \set {
    (\alpha, \beta) \in \tx{zip} \ y^2 \
    \middle| \
    \alpha \subseteq \beta
    }
    \right) \\
    &\set {
    \alpha \in \left(\cup\cup y\right)^{\cup\cup x}
    \mid
    \exists z \ (\alpha, z) - \tx{соответствие Галуа между $ x $ и $ y $}
    }
    \sim \\
    &\set {
    \alpha \in \left(\cup\cup x\right)^{\cup\cup y}
    \mid
    \exists z \ (z, \alpha) - \tx{соответствие Галуа между $ x $ и $ y $}
    } \\
    &\set {
    \alpha \in \mathcal{P} \ \tx{zip} \ x^2
    \mid
    \alpha - \tx{эквивалентность}
    }
    \sim
    \set{\alpha \in \mathcal{P} \ \mathcal{P} \ x \mid \alpha - \tx{разбиение $ x $}} \\
    &\set {
    (\alpha, \beta)
    \in
    \tx{zip}
    \set {
    \alpha \in \mathcal{P} \ \tx{zip} \ x^2 \
    \middle| \
    \alpha - \tx{эквивалентность}
    }^2 \
    \middle| \
    \alpha \subseteq \beta
    }
    -
    \tx{полная решётка}
\end{flalign*}

\subsection{Принцип трансфинитной индукции}
\begin{flalign*}
    x - \tx{фундированное}
    \tot
    \left(
    \forall y
    -
    \tx{$ \left(x \setminus \tx{idr}\cup\cup x\right) $-прогрессивное},
    z \in \cup\cup x \
    y \ z
    \right)
\end{flalign*}

\subsection{Цермело}
\begin{flalign*}
    \exists y - \tx{полное} \ x \equiv \cup\cup y
\end{flalign*}

\subsection{Цорна}
\begin{flalign*}
    &x - \tx{нестрогий порядок}
    \to
    \left(\forall y - \tx{$ x $-цепь} \ \exists\_ - \tx{$ x $-верхняя грань $ y $}\right)
    \to \\
    &\exists y \in \cup\cup x \
    y - \tx{$ \left(x \setminus \tx{idr}\cup\cup x\right) $-максимальный}
\end{flalign*}
